# Linux tarafı için Kurulum

git clone https://github.com/Hakan4178/base.git

cd base

chmod +x install.sh

./install.sh

# Kullanım

./run.sh           # Normal mod

./run.sh -d        # Debug mod (tüm paketleri loglar)

./run.sh -p 5000   # Farklı port
